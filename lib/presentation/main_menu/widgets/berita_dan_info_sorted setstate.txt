class _BeritaDanInfoWidgetState extends State<BeritaDanInfoWidget> {
  List<Item> beritaList = [];

  Future<void> fetchData() async {
    final response = await http
        .get(Uri.parse('${Constants.baseUrlStage}/publikasi/news/berita'));

    if (response.statusCode == 200) {
      List<dynamic> responseJson = json.decode(response.body);

      setState(() {
        beritaList = responseJson.map((e) => Item.fromJson(e)).toList();

        // Sorting beritaList berdasarkan tanggal
        beritaList.sort((a, b) => b.tanggal!.compareTo(a.tanggal!));
      });
    } else {
      throw Exception('Failed to load data from API');
    }
  }

  void initState() {
    //context.read<GetBeritaBloc>().add(DoGetBeritaSort());

    fetchData();
    super.initState();
  }

            child: Container(
              height: 190,
              width: MediaQuery.of(context).size.width * 0.9,
